1 OOP — Object Oriented Programming

Abstraction
Hides unnecessary internal details and exposes only what the user needs. You interact with a feature without knowing how it is implemented. 
public interface ICar { void Start(); }
public class Tesla : ICar {
    public void Start() =>
      Console.WriteLine("Started");
}
Encapsulation
Wraps data and methods inside a class and restricts direct access using access modifiers. 
public class BankAccount {
    private double balance;
    public void Deposit(double a)
        { balance += a; }
    public double GetBalance() => balance;
}
Inheritance
Allows a child class to acquire properties and methods from a parent class. Promotes code reuse. 
public class Animal {
    public void Eat() => Console.WriteLine("Eating");
}
public class Dog : Animal {
    public void Bark() => Console.WriteLine("Bark");
}
Polymorphism
"Many forms." Overloading (compile-time) and Overriding (runtime, child overrides parent). 
public class Animal {
    public virtual void Sound() => Console.WriteLine("...");
}
public class Dog : Animal {
    public override void Sound() => Console.WriteLine("Woof");
}

3 Practical C# Programs

String Programs

1. Reverse String

public static string ReverseString(string input)
{
    char[] arr = input.ToCharArray();
    int left = 0, right = arr.Length - 1;
    while (left < right) {
        char temp = arr[left];
        arr[left] = arr[right];
        arr[right] = temp;
        left++; right--;
    }
    return new string(arr);
}

2. Palindrome

public static bool IsPalindrome(string input)
{
    int left = 0, right = input.Length - 1;
    while (left < right) {
        if (char.ToLower(input[left]) != char.ToLower(input[right]))
            return false;
        left++; right--;
    }
    return true;
}

3. First Non-Repeating Character

public static char FirstNonRepeating(string input)
{
    var dict = new Dictionary<char, int>();
    foreach (char c in input)
        dict[c] = dict.ContainsKey(c) ? dict[c] + 1 : 1;
    foreach (char c in input)
        if (dict[c] == 1) return c;
    return '\0';
}

4. Reverse Words in Sentence

public static string ReverseWords(string input)
{
    var words = input.Split(' ',
      StringSplitOptions.RemoveEmptyEntries);
    int left = 0, right = words.Length - 1;
    while (left < right) {
        (words[left], words[right]) = (words[right], words[left]);
        left++; right--;
    }
    return string.Join(" ", words);
}

5. Find Duplicates

public static void FindDuplicates(int[] arr)
{
    var seen = new HashSet<int>();
    var dups = new HashSet<int>();
    foreach (int num in arr)
        if (!seen.Add(num)) dups.Add(num);
    foreach (var d in dups) Console.WriteLine(d);
}

6. Second Largest

public static int SecondLargest(int[] arr)
{
    int first = int.MinValue, second = int.MinValue;
    foreach (int n in arr) {
        if (n > first) { second = first; first = n; }
        else if (n > second && n != first) second = n;
    }
    return second;
}

7. Move Zeros to End

public static void MoveZeros(int[] arr)
{
    int index = 0;
    for (int i = 0; i < arr.Length; i++)
        if (arr[i] != 0) arr[index++] = arr[i];
    while (index < arr.Length) arr[index++] = 0;
}

8. Rotate Array

public static void Rotate(int[] arr, int k)
{
    k = k % arr.Length;
    Array.Reverse(arr, 0, arr.Length);
    Array.Reverse(arr, 0, k);
    Array.Reverse(arr, k, arr.Length - k);
}

LRU Cache

class LRUCache
{
    private int capacity;
    private Dictionary<int, LinkedListNode<(int key, int value)>> dict;
    private LinkedList<(int key, int value)> list;

    public LRUCache(int capacity) {
        this.capacity = capacity;
        dict = new(); list = new();
    }

    public int Get(int key) {
        if (!dict.ContainsKey(key)) return -1;
        var node = dict[key];
        list.Remove(node); list.AddFirst(node);
        return node.Value.value;
    }

    public void Put(int key, int value) {
        if (dict.ContainsKey(key)) {
            list.Remove(dict[key]);
        } else if (dict.Count == capacity) {
            var last = list.Last;
            dict.Remove(last.Value.key);
            list.RemoveLast();
        }
        var newNode = new LinkedListNode<(int, int)>((key, value));
        list.AddFirst(newNode);
        dict[key] = newNode;
    }
}

5 Middleware

public class LoggingMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ILogger<LoggingMiddleware> _logger;

    public LoggingMiddleware(RequestDelegate next, ILogger<LoggingMiddleware> logger)
    { _next = next; _logger = logger; }

    public async Task Invoke(HttpContext context)
    {
        var watch = Stopwatch.StartNew();
        _logger.LogInformation("Request: {Method} {Path}",
            context.Request.Method, context.Request.Path);

        await _next(context);   // ← Pass to next middleware

        watch.Stop();
        _logger.LogInformation("Response: {StatusCode} in {Ms}ms",
            context.Response.StatusCode, watch.ElapsedMilliseconds);
    }
}
// Register in Program.cs

8 JWT Authentication

JWT (JSON Web Token) is a compact, self-contained token. After login, the server issues a JWT containing the user's ID and roles. The client sends it as Authorization: Bearer <token> on every API call. The server validates the signature — no database lookup needed on every request, making it stateless and scalable.

// Generate JWT
public string GenerateToken(User user)
{
    var key    = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(_config["Jwt:Secret"]));
    var creds  = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);
    var claims = new[] {
        new Claim(JwtRegisteredClaimNames.Sub, user.Id.ToString()),
        new Claim(ClaimTypes.Email, user.Email),
        new Claim(ClaimTypes.Role, user.Role),
        new Claim(JwtRegisteredClaimNames.Jti, Guid.NewGuid().ToString())
    };
    var token = new JwtSecurityToken(
        issuer: _config["Jwt:Issuer"], audience: _config["Jwt:Audience"],
        claims: claims, expires: DateTime.UtcNow.AddMinutes(60),
        signingCredentials: creds);
    return new JwtSecurityTokenHandler().WriteToken(token);
}
Security Trap Never store JWT tokens in localStorage — XSS attacks can steal them. Store access tokens in memory (Angular service) and refresh tokens in HttpOnly cookies.
Power Line "We use 15-minute JWT access tokens with 30-day refresh tokens in HttpOnly cookies. The Angular app intercepts 401 responses via an HTTP interceptor, silently calls the refresh endpoint, and retries the original request — the user never sees a logout."

10D Webhooks

A webhook is an HTTP callback — instead of polling every few seconds, the external service calls YOUR endpoint the moment something happens. Event-driven, push-based communication.

[HttpPost("stripe")]
public async Task<IActionResult> StripeWebhook()
{
    var json = await new StreamReader(Request.Body).ReadToEndAsync();
    var sig  = Request.Headers["Stripe-Signature"];
    try
    {
        // Step 1: Verify signature — reject if invalid
        var stripeEvent = EventUtility.ConstructEvent(
            json, sig, _config["Stripe:WebhookSecret"]);

        // Step 2: Handle event type
        if (stripeEvent.Type == Events.PaymentIntentSucceeded)
        {
            var pi = stripeEvent.Data.Object as PaymentIntent;
            await _backgroundQueue.EnqueueAsync(new OrderPaidJob(pi.Id));
        }

        // Step 3: Return 200 immediately — Stripe retries if you don't
        return Ok();
    }
    catch (StripeException) { return BadRequest(); }
}
4 Rules (1) Always verify signature. (2) Return 200 fast. (3) Make handler idempotent. (4) Enqueue heavy work to background queue.

-- AFTER UPDATE Trigger — Track Price Changes
CREATE OR ALTER TRIGGER trg_Product_PriceChange
ON Product
AFTER UPDATE
AS
BEGIN
    SET NOCOUNT ON;
    IF UPDATE(Price)
    BEGIN
        INSERT INTO PriceHistory (ProductId, OldPrice, NewPrice, ChangedAt)
        SELECT d.Id, d.Price, i.Price, GETDATE()
        FROM DELETED d                         -- old values
        INNER JOIN INSERTED i ON d.Id = i.Id   -- new values
        WHERE d.Price <> i.Price;
    END
END

CREATE OR ALTER TRIGGER trg_Customer_PreventDelete ON Customer
INSTEAD OF DELETE AS
BEGIN

    WHERE Id IN (SELECT Id FROM DELETED);
END

    @customerId INT, @purchasedItems Type_Order_Item READONLY
AS BEGIN
    SET NOCOUNT ON;
    BEGIN TRY
        BEGIN TRANSACTION
            INSERT INTO Orders (CustomerId, OrderDate) VALUES (@customerId, GETDATE());

            INSERT INTO OrderItem (OrderId, ItemId, Price, Quantity)

export class AuthInterceptor implements HttpInterceptor {
    intercept(req: HttpRequest<any>, next: HttpHandler): Observable<HttpEvent<any>> {
        const token = this.auth.getToken();
        const authReq = token
            ? req.clone({ setHeaders: { Authorization: `Bearer ${token}` } })
            : req;

        return next.handle(authReq).pipe(
            catchError((err: HttpErrorResponse) => {
                if (err.status === 401) { this.auth.logout(); this.router.navigate(['/login']); }
                return throwError(() => err);
            })
        );
    }
}

29 Common Interview Traps

⚠ Trap 1 Scoped inside Singleton — captive dependency causes stale data. Use IServiceScopeFactory.
⚠ Trap 2 IEnumerable for DB filtering — loads entire table into memory. Use IQueryable.
⚠ Trap 3 Missing await — method returns a Task nobody awaits; exception swallowed silently.
⚠ Trap 4 async void — exceptions unobservable, crash the process. Always use async Task.
⚠ Trap 5 SELECT * — fetches unnecessary columns, breaks on schema change. Always select specific columns.
⚠ Trap 6 No CancellationToken — long operations run even after client disconnects.
⚠ Trap 7 Lazy loading in APIs — silent N+1 queries. Use eager loading with Include().
⚠ Trap 8 NOT IN with NULL subquery — silently returns zero rows. Use NOT EXISTS.
