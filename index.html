1 OOP — Object Oriented Programming

Abstraction
Hides unnecessary internal details and exposes only what the user needs. You interact with a feature without knowing how it is implemented. 
public interface ICar { void Start(); }
public class Tesla : ICar {
    public void Start() =>
      Console.WriteLine("Started");
}
Encapsulation
Wraps data and methods inside a class and restricts direct access using access modifiers. 
public class BankAccount {
    private double balance;
    public void Deposit(double amount)
        { balance += amount; }
    public double GetBalance() => balance;
}
Inheritance
Allows a child class to acquire properties and methods from a parent class. Promotes code reuse. 
public class Animal {
    public void Eat() => Console.WriteLine("Eating");
}
public class Dog : Animal {
    public void Bark() => Console.WriteLine("Bark");
}
Polymorphism
"Many forms." Overloading (compile-time, same name different params) and Overriding (runtime, child overrides parent). 
public class Animal {
    public virtual void Sound() => Console.WriteLine("...");
}
public class Dog : Animal {
    public override void Sound() => Console.WriteLine("Woof");
}

3 Practical C# Programs

String Programs

1. Reverse String (No built-in)

public static string ReverseString(string input)
{
    char[] arr = input.ToCharArray();
    int left = 0, right = arr.Length - 1;
    while (left < right) {
        char temp = arr[left];
        arr[left] = arr[right];
        arr[right] = temp;
        left++; right--;
    }
    return new string(arr);
}

2. Palindrome (Optimized)

public static bool IsPalindrome(string input)
{
    int left = 0, right = input.Length - 1;
    while (left < right) {
        if (char.ToLower(input[left]) != char.ToLower(input[right]))
            return false;
        left++; right--;
    }
    return true;
}

3. First Non-Repeating Character

public static char FirstNonRepeating(string input)
{
    var dict = new Dictionary<char, int>();
    foreach (char c in input)
        dict[c] = dict.ContainsKey(c) ? dict[c] + 1 : 1;
    foreach (char c in input)
        if (dict[c] == 1) return c;
    return '\0';
}

4. Reverse Words in Sentence

public static string ReverseWords(string input)
{
    var words = input.Split(' ',
      StringSplitOptions.RemoveEmptyEntries);
    int left = 0, right = words.Length - 1;
    while (left < right) {
        (words[left], words[right]) = (words[right], words[left]);
        left++; right--;
    }
    return string.Join(" ", words);
}

5. Find Duplicates

public static void FindDuplicates(int[] arr)
{
    var seen = new HashSet<int>();
    var dups = new HashSet<int>();
    foreach (int num in arr)
        if (!seen.Add(num)) dups.Add(num);
    foreach (var d in dups) Console.WriteLine(d);

6. Second Largest (No sorting)

public static int SecondLargest(int[] arr)
{
    int first = int.MinValue, second = int.MinValue;
    foreach (int n in arr) {
        if (n > first) { second = first; first = n; }
        else if (n > second && n != first) second = n;
    }
    return second;
}

7. Move Zeros to End

public static void MoveZeros(int[] arr)
{
    int index = 0;
    for (int i = 0; i < arr.Length; i++)
        if (arr[i] != 0) arr[index++] = arr[i];
    while (index < arr.Length) arr[index++] = 0;
}

8. Rotate Array

public static void Rotate(int[] arr, int k)
{
    k = k % arr.Length;
    Array.Reverse(arr, 0, arr.Length);
    Array.Reverse(arr, 0, k);
    Array.Reverse(arr, k, arr.Length - k);
}

11. Group Anagrams

public static void GroupAnagrams(string[] words)
{
    var dict = new Dictionary<string, List<string>>();
    foreach (var word in words) {
        var key = new string(word.OrderBy(c => c).ToArray());
        if (!dict.ContainsKey(key))
            dict[key] = new List<string>();
        dict[key].Add(word);
    }
    foreach (var group in dict.Values)
        Console.WriteLine(string.Join(", ", group));
}

LRU Cache (Important)

class LRUCache
{
    private int capacity;
    private Dictionary<int, LinkedListNode<(int key, int value)>> dict;
    private LinkedList<(int key, int value)> list;

    public LRUCache(int capacity) {
        this.capacity = capacity;
        dict = new(); list = new();
    }

    public int Get(int key) {
        if (!dict.ContainsKey(key)) return -1;
        var node = dict[key];
        list.Remove(node); list.AddFirst(node);
        return node.Value.value;
    }

    public void Put(int key, int value) {
        if (dict.ContainsKey(key)) {
            list.Remove(dict[key]);
        } else if (dict.Count == capacity) {
            var last = list.Last;
            dict.Remove(last.Value.key);
            list.RemoveLast();
        }
        var newNode = new LinkedListNode<(int, int)>((key, value));
        list.AddFirst(newNode);
        dict[key] = newNode;
    }
}

5 Middleware

Custom Logging Middleware

public class LoggingMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ILogger<LoggingMiddleware> _logger;

    public LoggingMiddleware(RequestDelegate next, ILogger<LoggingMiddleware> logger)
    { _next = next; _logger = logger; }

    public async Task Invoke(HttpContext context)
    {
        var watch = Stopwatch.StartNew();
        _logger.LogInformation("Request: {Method} {Path}",
            context.Request.Method, context.Request.Path);

        await _next(context);   // ← Pass to next middleware

        watch.Stop();
        _logger.LogInformation("Response: {StatusCode} in {Ms}ms",
            context.Response.StatusCode, watch.ElapsedMilliseconds);
    }
}
// Register in Program.cs
app.UseMiddleware<LoggingMiddleware>();

7 Async / Await & Task

Key Mental Model: await releases the thread back to the pool while waiting for I/O. It does NOT create a new thread. More concurrent users with the same thread count.

Parallel Execution

// ❌ Sequential — slow (total = t1 + t2 + t3)

{

    {

Window Functions — Complete Guide

Window functions compute a value across a window (set of rows related to the current row) without collapsing rows like GROUP BY does. Syntax: FUNCTION() OVER (PARTITION BY ... ORDER BY ...)

Ranking Functions

FunctionHandles TiesGaps After Tie?Example Result (scores: 100,100,90)
ROW_NUMBER()Arbitrary unique number—1, 2, 3
RANK()Same rank for tied rowsYes — skips next rank1, 1, 3
DENSE_RANK()Same rank for tied rowsNo — next rank is sequential1, 1, 2
NTILE(n)Divides rows into n equal buckets—Bucket 1, 1, 2 (for NTILE(2))
-- ROW_NUMBER: unique number per partition — used to get latest record per group
WITH ranked AS (
    SELECT *, ROW_NUMBER() OVER (PARTITION BY CustomerId ORDER BY OrderDate DESC) AS rn
    FROM Orders
)
SELECT * FROM ranked WHERE rn = 1;   -- latest order per customer

-- RANK vs DENSE_RANK — side by side comparison
SELECT
    Name, Score,
    RANK()       OVER (ORDER BY Score DESC) AS rnk,        -- 1,1,3
    DENSE_RANK() OVER (ORDER BY Score DESC) AS dense_rnk,  -- 1,1,2
    ROW_NUMBER() OVER (ORDER BY Score DESC) AS row_num      -- 1,2,3
FROM Students;

-- NTILE(4): divide rows into 4 quartiles for performance bands
SELECT Name, Salary,
    NTILE(4) OVER (ORDER BY Salary DESC) AS Quartile
FROM Employee;

-- LAG: compare each month's sales to previous month
SELECT
    FORMAT(OrderDate,'yyyy-MM') AS Month,
    SUM(Amount) AS Sales,
    LAG(SUM(Amount), 1, 0) OVER (ORDER BY FORMAT(OrderDate,'yyyy-MM')) AS PrevMonthSales,
    SUM(Amount) - LAG(SUM(Amount), 1, 0) OVER (ORDER BY FORMAT(OrderDate,'yyyy-MM')) AS Diff
FROM Orders
GROUP BY FORMAT(OrderDate,'yyyy-MM');

    LEAD(OrderDate, 1, NULL) OVER (PARTITION BY CustomerId ORDER BY OrderDate) AS NextOrderDate
FROM Orders;

-- FIRST_VALUE / LAST_VALUE: cheapest + most expensive item per category
SELECT Name, Category, Price,
    FIRST_VALUE(Price) OVER (PARTITION BY Category ORDER BY Price ASC)  AS CheapestInCat,
    LAST_VALUE(Price)  OVER (PARTITION BY Category ORDER BY Price ASC
        ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING)       AS MostExpensiveInCat
FROM Product;
LAST_VALUE Trap By default, the window frame is ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW — so LAST_VALUE just returns the current row's value. Always add ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING to get the true last value.

23B SQL Triggers

A trigger is a special stored procedure that automatically fires before or after a DML event (INSERT, UPDATE, DELETE) on a table. You cannot call a trigger manually — SQL Server fires it automatically.

TypeWhen It FiresCommon Use
AFTER (FOR)After DML completesAudit logging, cascade updates
INSTEAD OFIn place of DMLUpdatable views, custom logic replacing default action
DDL TriggerOn schema changes (CREATE/ALTER/DROP)Prevent accidental schema drops in prod
Magic Tables Inside a trigger, SQL Server provides two virtual tables: INSERTED (new/updated rows) and DELETED (old/deleted rows). These are your primary tool for reading what changed.

AFTER INSERT Trigger — Audit Log

-- Fires every time a row is inserted into Orders
CREATE OR ALTER TRIGGER trg_Orders_AfterInsert
ON Orders
AFTER INSERT
AS
BEGIN
    SET NOCOUNT ON;
    INSERT INTO AuditLog (TableName, Action, RecordId, ChangedAt)
    SELECT 'Orders', 'INSERT', Id, GETDATE()
    FROM INSERTED;   -- INSERTED holds the newly added row(s)
END

AFTER UPDATE Trigger — Track Price Changes

CREATE OR ALTER TRIGGER trg_Product_PriceChange
ON Product
AFTER UPDATE
AS
BEGIN
    SET NOCOUNT ON;
    -- Only fire if Price column was actually changed
    IF UPDATE(Price)
    BEGIN
        INSERT INTO PriceHistory (ProductId, OldPrice, NewPrice, ChangedAt)
        SELECT d.Id, d.Price, i.Price, GETDATE()
        FROM DELETED d                  -- old values
        INNER JOIN INSERTED i ON d.Id = i.Id   -- new values
        WHERE d.Price <> i.Price;
    END
END

INSTEAD OF Trigger — Prevent Deletion

CREATE OR ALTER TRIGGER trg_Customer_PreventDelete
ON Customer
INSTEAD OF DELETE
AS
BEGIN
    SET NOCOUNT ON;
    -- Soft-delete instead of hard delete
    UPDATE Customer
    SET IsDeleted = 1, DeletedAt = GETDATE()
    WHERE Id IN (SELECT Id FROM DELETED);
END

DDL Trigger — Prevent Table Drops

CREATE TRIGGER trg_PreventDrop
ON DATABASE
FOR DROP_TABLE, ALTER_TABLE
AS
BEGIN
    PRINT 'Schema changes are not allowed in production!';
    ROLLBACK;
END

23C SQL Performance Tuning

Performance tuning is the process of finding and fixing slow queries. The main tools are Execution Plans, indexes, and query rewriting.

Step 1 — Find the Slow Query

-- Top 10 most expensive queries by total CPU (SQL Server)
SELECT TOP 10
    qs.total_elapsed_time / qs.execution_count AS avg_elapsed_us,
    qs.execution_count,
    SUBSTRING(qt.text, (qs.statement_start_offset/2)+1, 200) AS query_text
FROM sys.dm_exec_query_stats qs
CROSS APPLY sys.dm_exec_sql_text(qs.sql_handle) qt
ORDER BY avg_elapsed_us DESC;

Step 2 — Read the Execution Plan

OperatorMeaningAction
Table ScanNo index — reads every rowAdd an index on filtered/joined column
Index ScanIndex exists but reading all rows in itAdd a more selective index or covering index
Index SeekEfficient — jumps straight to matching rowsGood — aim for this
Key LookupIndex found row but fetches extra columns from heapUse INCLUDE columns in index
Hash Join / SortMemory-heavy operationsCan indicate missing index on join column

Step 3 — Indexing Strategy

-- Covering index: OrderId is the seek key, extra columns come free
CREATE NONCLUSTERED INDEX IX_OrderItem_Covering
ON OrderItem(OrderId)
INCLUDE (ItemId, Quantity, Price);   -- eliminates Key Lookup

-- Composite index: column order matters — put equality first, range last
CREATE NONCLUSTERED INDEX IX_Orders_Customer_Date
ON Orders(CustomerId, OrderDate DESC);
-- Good for: WHERE CustomerId = 5 ORDER BY OrderDate DESC

Stored Procedure — Place Order with Transaction

CREATE OR ALTER PROC PlaceOrder
    @customerId    INT,
    @purchasedItems Type_Order_Item READONLY
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRY
        IF @customerId IS NULL OR (SELECT COUNT(*) FROM @purchasedItems) < 1
            RETURN;

        DECLARE @NewOrderId INT;

        BEGIN TRANSACTION
            INSERT INTO Orders (CustomerId, OrderDate)
            VALUES (@customerId, GETDATE());

            SET @NewOrderId = SCOPE_IDENTITY();

            INSERT INTO OrderItem (OrderId, ItemId, Price, Quantity)
            SELECT @NewOrderId, ItemId, Price, Qty FROM @purchasedItems
        COMMIT TRANSACTION
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END

HTTP Interceptor — JWT + Error Handling

// Intercept every HTTP request — add token, handle 401
@Injectable()
export class AuthInterceptor implements HttpInterceptor {
    constructor(private auth: AuthService, private router: Router) {}

    intercept(req: HttpRequest<any>, next: HttpHandler): Observable<HttpEvent<any>> {
        const token = this.auth.getToken();

        const authReq = token
            ? req.clone({ setHeaders: { Authorization: `Bearer ${token}` } })
            : req;

        return next.handle(authReq).pipe(
            catchError((err: HttpErrorResponse) => {
                if (err.status === 401) {
                    this.auth.logout();
                    this.router.navigate(['/login']);
                }
                if (err.status === 0) {
                    console.error('Network error');
                }
                return throwError(() => err);
            })
        );
    }
}

30A Git & Version Control

Git is the industry-standard distributed version control system. Every commit is a snapshot of your entire project, not a diff. You have a full local copy of the repository.

Core Concepts

ConceptDescription
Working DirectoryFiles you're currently editing. Changes are "unstaged" until you git add them.
Staging Area (Index)Snapshot of changes ready to be committed. git add file.cs moves changes here.
Local RepositoryThe full history on your machine. git commit writes to this.
Remote RepositoryShared repo (GitHub, Azure DevOps). git push sends commits here.
BranchAn independent line of development. Cheap and fast in Git. main is the default.
HEADPointer to your current position (usually the latest commit on your branch).

Essential Commands

# Daily workflow
git status                    # see what's changed
git add .                     # stage all changes
git add src/Controllers/      # stage specific folder
git commit -m "feat: add user endpoint"
git push origin feature/user-api

# Branching
git checkout -b feature/payment  # create + switch to new branch
git checkout main                # switch back
git merge feature/payment        # merge into current branch
git branch -d feature/payment    # delete merged branch

# Syncing with remote
git fetch origin      # download changes, don't apply yet
git pull origin main  # fetch + merge in one step
git push origin main

# Viewing history
git log --oneline --graph    # compact history with branch graph
git diff HEAD~1              # what changed in last commit

Merge vs Rebase

CommandWhat It DoesWhen to Use
git mergeCombines branches, creates a merge commit. Preserves full history.Merging feature branches into main. Use in team environments.
git rebaseReplays your commits on top of target branch. Linear history, no merge commit.Cleaning up local commits before PR. Never rebase shared branches.
git cherry-pickApply a specific commit from another branch.Hotfix: apply a bug fix to multiple release branches.

Undoing Changes

# Undo staged changes (unstage file)
git restore --staged src/Service.cs

# Undo uncommitted working directory changes
git restore src/Service.cs

# Undo last commit but keep changes staged
git reset --soft HEAD~1

# Undo last commit and discard changes (DESTRUCTIVE)
git reset --hard HEAD~1

# Revert a commit safely (creates a new commit undoing it)
git revert abc1234     # safe for shared branches
⚠ Rule Never force push (git push --force) to a shared branch like main. It rewrites history for everyone and causes data loss. Use git revert instead to safely undo changes on shared branches.

29 Common Interview Traps

⚠ Trap 1 Scoped inside Singleton — captive dependency causes stale data or runtime errors. Use IServiceScopeFactory.
⚠ Trap 2 IEnumerable for DB filtering — loads entire table into memory, then filters in C#. Use IQueryable.
⚠ Trap 3 Missing await — method returns a Task that nobody awaits; exception swallowed silently.
⚠ Trap 4 async void — exceptions are unobservable and crash the process. Always use async Task.
⚠ Trap 5 SELECT * — fetches unnecessary columns, breaks if schema changes. Always select specific columns.
⚠ Trap 6 No CancellationToken — long operations run even after client disconnects. Always accept and pass CancellationToken.
⚠ Trap 7 Lazy loading in APIs — causes N+1 queries silently, only noticed under load. Use eager loading with Include().
