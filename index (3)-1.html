1 OOP — Object Oriented Programming

Abstraction
Hides unnecessary internal details and exposes only what the user needs. You interact with a feature without knowing how it is implemented. 
public interface ICar { void Start(); }
public class Tesla : ICar {
    public void Start() =>
      Console.WriteLine("Started");
}
Encapsulation
Wraps data and methods inside a class and restricts direct access using access modifiers. 
public class BankAccount {
    private double balance;
    public void Deposit(double amount)
        { balance += amount; }
    public double GetBalance() => balance;
}
Inheritance
Allows a child class to acquire properties and methods from a parent class. Promotes code reuse. 
public class Animal {
    public void Eat() => Console.WriteLine("Eating");
}
public class Dog : Animal {
    public void Bark() => Console.WriteLine("Bark");
}
Polymorphism
"Many forms." Overloading (compile-time, same name different params) and Overriding (runtime, child overrides parent). 
public class Animal {
    public virtual void Sound() => Console.WriteLine("...");
}
public class Dog : Animal {
    public override void Sound() => Console.WriteLine("Woof");
}

String Programs

1. Reverse String (No built-in)

public static string ReverseString(string input)
{
    char[] arr = input.ToCharArray();
    int left = 0, right = arr.Length - 1;
    while (left < right) {
        char temp = arr[left];
        arr[left] = arr[right];
        arr[right] = temp;
        left++; right--;
    }
    return new string(arr);
}

2. Palindrome (Optimized)

public static bool IsPalindrome(string input)
{
    int left = 0, right = input.Length - 1;
    while (left < right) {
        if (char.ToLower(input[left]) != char.ToLower(input[right]))
            return false;
        left++; right--;
    }
    return true;
}

3. First Non-Repeating Character

public static char FirstNonRepeating(string input)
{
    var dict = new Dictionary<char, int>();
    foreach (char c in input)
        dict[c] = dict.ContainsKey(c) ? dict[c] + 1 : 1;
    foreach (char c in input)
        if (dict[c] == 1) return c;
    return '\0';
}

4. Reverse Words in Sentence

public static string ReverseWords(string input)
{
    var words = input.Split(' ',
      StringSplitOptions.RemoveEmptyEntries);
    int left = 0, right = words.Length - 1;
    while (left < right) {
        (words[left], words[right]) = (words[right], words[left]);
        left++; right--;
    }
    return string.Join(" ", words);
}

6. Second Largest (No sorting)

public static int SecondLargest(int[] arr)
{
    int first = int.MinValue, second = int.MinValue;
    foreach (int n in arr) {
        if (n > first) { second = first; first = n; }
        else if (n > second && n != first) second = n;
    }
    return second;
}

7. Move Zeros to End

public static void MoveZeros(int[] arr)
{
    int index = 0;
    for (int i = 0; i < arr.Length; i++)
        if (arr[i] != 0) arr[index++] = arr[i];
    while (index < arr.Length) arr[index++] = 0;
}

8. Rotate Array

public static void Rotate(int[] arr, int k)
{
    k = k % arr.Length;
    Array.Reverse(arr, 0, arr.Length);
    Array.Reverse(arr, 0, k);
    Array.Reverse(arr, k, arr.Length - k);
}

LRU Cache (Important)

class LRUCache
{
    private int capacity;
    private Dictionary<int, LinkedListNode<(int key, int value)>> dict;
    private LinkedList<(int key, int value)> list;

    public LRUCache(int capacity) {
        this.capacity = capacity;
        dict = new(); list = new();
    }

    public int Get(int key) {
        if (!dict.ContainsKey(key)) return -1;
        var node = dict[key];
        list.Remove(node); list.AddFirst(node);
        return node.Value.value;
    }

    public void Put(int key, int value) {
        if (dict.ContainsKey(key)) {
            list.Remove(dict[key]);
        } else if (dict.Count == capacity) {
            var last = list.Last;
            dict.Remove(last.Value.key);
            list.RemoveLast();
        }
        var newNode = new LinkedListNode<(int, int)>((key, value));
        list.AddFirst(newNode);
        dict[key] = newNode;
    }
}

public class LoggingMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ILogger<LoggingMiddleware> _logger;

    public LoggingMiddleware(RequestDelegate next, ILogger<LoggingMiddleware> logger)
    { _next = next; _logger = logger; }

    public async Task Invoke(HttpContext context)
    {
        var watch = Stopwatch.StartNew();
        _logger.LogInformation("Request: {Method} {Path}",
            context.Request.Method, context.Request.Path);

        await _next(context);   // ← Pass to next middleware

        watch.Stop();
        _logger.LogInformation("Response: {StatusCode} in {Ms}ms",
            context.Response.StatusCode, watch.ElapsedMilliseconds);
    }
}
// Register in Program.cs
app.UseMiddleware<LoggingMiddleware>();

8 JWT Authentication

JWT (JSON Web Token) is a compact, self-contained token used to securely transmit information between parties as a JSON object. After a user logs in, the server issues a JWT. The client includes it in every subsequent request. The server validates the token's signature — no database lookup needed on every request, making it stateless and scalable.

Real-Life Scenario: Your Angular app calls POST /api/auth/login with username/password. Your .NET API validates credentials, creates a JWT containing the user's ID and roles, and returns it. Angular stores the token and sends it as Authorization: Bearer <token> on every API call. The API validates the signature and grants or denies access — no session, no server-side state, works across 10 API instances identically.

JWT Structure — Header.Payload.Signature

A JWT has three Base64url-encoded parts separated by dots: xxxxx.yyyyy.zzzzz

  • Header: Algorithm type ({"alg":"HS256","typ":"JWT"})
  • Payload (Claims): User data (sub=userId, role, exp=expiry, iat=issued-at)
  • Signature: HMAC-SHA256(Base64(header) + "." + Base64(payload), secretKey) — this verifies the token wasn't tampered with
// .NET 8 — Generate JWT token
public string GenerateToken(User user)
{
    var key     = new SymmetricSecurityKey(Encoding.UTF8.GetBytes(_config["Jwt:Secret"]));
    var creds   = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);
    var claims  = new[]
    {
        new Claim(JwtRegisteredClaimNames.Sub, user.Id.ToString()),
        new Claim(ClaimTypes.Email, user.Email),
        new Claim(ClaimTypes.Role, user.Role),   // "Admin", "User", etc.
        new Claim(JwtRegisteredClaimNames.Jti, Guid.NewGuid().ToString())
    };

    var token = new JwtSecurityToken(
        issuer:   _config["Jwt:Issuer"],
        audience: _config["Jwt:Audience"],
        claims:   claims,
        expires:  DateTime.UtcNow.AddMinutes(60),
        signingCredentials: creds
    );

    return new JwtSecurityTokenHandler().WriteToken(token);
}

// Program.cs — Validate JWT on incoming requests
builder.Services.AddAuthentication(JwtBearerDefaults.AuthenticationScheme)
    .AddJwtBearer(options => {
        options.TokenValidationParameters = new TokenValidationParameters {
            ValidateIssuer           = true,
            ValidateAudience         = true,
            ValidateLifetime         = true,
            ValidateIssuerSigningKey = true,
            ValidIssuer              = config["Jwt:Issuer"],
            ValidAudience            = config["Jwt:Audience"],
            IssuerSigningKey         = new SymmetricSecurityKey(
                                           Encoding.UTF8.GetBytes(config["Jwt:Secret"]))
        };
    });

// Controller — require auth and check roles
[Authorize]                        // any authenticated user
[Authorize(Roles = "Admin")]       // only Admin role
public class OrderController : ControllerBase { ... }

10C FluentValidation

FluentValidation is a popular .NET library for building strongly-typed validation rules using a fluent, readable API. Instead of cluttering your model with data annotation attributes ([Required], [MaxLength]), you write dedicated validator classes — keeping models clean and validation logic testable, reusable, and separate.

Real-Life Scenario: You have an API endpoint that creates a customer. Without FluentValidation you'd litter the DTO with attributes and duplicate logic everywhere. With FluentValidation, you have one CreateCustomerValidator class that is unit-testable and reusable across controllers and command handlers (CQRS).

Installation & Basic Usage

// Install: dotnet add package FluentValidation.AspNetCore

// DTO — clean, no annotations needed
public class CreateCustomerRequest
{
    public string Name    { get; set; }
    public string Email   { get; set; }
    public int    Age     { get; set; }
    public string Phone   { get; set; }
}

// Validator — all rules in one place
public class CreateCustomerValidator : AbstractValidator<CreateCustomerRequest>
{
    public CreateCustomerValidator()
    {
        RuleFor(x => x.Name)
            .NotEmpty().WithMessage("Name is required.")
            .MinimumLength(2).WithMessage("Name must be at least 2 characters.")
            .MaximumLength(100);

        RuleFor(x => x.Email)
            .NotEmpty()

10D Webhooks — How They Work

A webhook is an HTTP callback — instead of your app polling an external service every few seconds asking "did anything happen?", the external service calls YOUR endpoint the moment something happens. It is event-driven, push-based communication.

Analogy: Polling is like calling a restaurant every 5 minutes asking "is my order ready?" A webhook is the restaurant calling you when it's ready.

You register a URL with the service (e.g. Stripe)
→
Event occurs (payment succeeds)
→
Service sends HTTP POST to your URL
→
Your endpoint receives JSON payload
→
Verify signature
→
Process event & return 200 OK fast
→
Do heavy work asynchronously (queue)

.NET Webhook Endpoint — Stripe Example

// CRITICAL RULES:
// 1. Always verify the webhook signature before trusting the payload
// 2. Return 200 OK immediately — do NOT do heavy work synchronously
// 3. Enqueue heavy work (send email, update DB) to a background queue
// 4. Make the handler idempotent — Stripe may send the same event twice

[ApiController]
[Route("api/webhooks")]
public class WebhookController : ControllerBase
{
    [HttpPost("stripe")]
    public async Task<IActionResult> StripeWebhook()
    {
        var json = await new StreamReader(Request.Body).ReadToEndAsync();
        var sig  = Request.Headers["Stripe-Signature"];

        try
        {
            // Step 1: Verify signature — reject if invalid
            var stripeEvent = EventUtility.ConstructEvent(
                json, sig, _config["Stripe:WebhookSecret"]);

            // Step 2: Handle event type
            switch (stripeEvent.Type)
            {
                case Events.PaymentIntentSucceeded:
                    var pi = stripeEvent.Data.Object as PaymentIntent;
                    // Enqueue to background job — don't block here!
                    await _backgroundQueue.EnqueueAsync(new OrderPaidJob(pi.Id));
                    break;

                case Events.InvoicePaymentFailed:
                    var inv = stripeEvent.Data.Object as Invoice;
                    await _backgroundQueue.EnqueueAsync(new SuspendSubscriptionJob(inv.CustomerId));
                    break;
            }

            // Step 3: Return 200 immediately — Stripe retries if you don't
            return Ok();
        }
        catch (StripeException ex)
        {
            _logger.LogWarning("Invalid Stripe webhook: {Msg}", ex.Message);
            return BadRequest();
        }
    }
}

Webhook Security — Must-Know

Verify Signature
Every serious webhook provider sends a signature (HMAC-SHA256). Always verify it. Never trust the payload without this check — anyone can POST to your URL.
Idempotency
Webhooks can be delivered more than once (network retry). Use the event ID to check if you've already processed it. Storing processed event IDs prevents double-charging or double-emails.
Return 200 Fast
If you take too long, the provider thinks delivery failed and retries. Process synchronously only what's needed to validate, then enqueue everything else.
Use HTTPS Only
Never expose a webhook endpoint over plain HTTP. Signatures can be replayed if traffic is intercepted without encryption.

Correlation ID

Definition: A Correlation ID is a unique identifier (GUID) assigned to every incoming HTTP request and passed through all downstream service calls and log entries, so you can trace the complete journey of a single request across multiple services and logs.

Real-Life Scenario: A user reports "my payment failed at 2:47 PM." You take the Correlation ID from their browser's network tab (or from the error response header), search your centralized logs in Application Insights / Seq for that ID, and instantly see every log entry — from the Angular frontend call → .NET API → Payment Service → Stripe API call — in sequence, in one view. Without it, you're searching through millions of log lines with no thread to follow.

// Middleware — assign CorrelationId on every request
public class CorrelationIdMiddleware
{
    public async Task Invoke(HttpContext context)
    {
        var correlationId = context.Request.Headers["X-Correlation-Id"].FirstOrDefault()
                            ?? Guid.NewGuid().ToString();

        context.Response.Headers["X-Correlation-Id"] = correlationId;

        using (_logger.BeginScope(new Dictionary<string, object> {
            ["CorrelationId"] = correlationId }))
        {
            await _next(context);   // all logs within this scope include CorrelationId
        }
    }
}

Window Functions — Complete Guide

Window functions compute a value across a window (set of rows related to the current row) without collapsing rows like GROUP BY does. Syntax: FUNCTION() OVER (PARTITION BY ... ORDER BY ...)

WITH ranked AS (
    SELECT *, ROW_NUMBER() OVER (PARTITION BY CustomerId ORDER BY OrderDate DESC) AS rn
    FROM Orders
)
SELECT * FROM ranked WHERE rn = 1;   -- latest order per customer

    Name, Score,
    RANK()       OVER (ORDER BY Score DESC) AS rnk,        -- 1,1,3

FROM Students;

This is one of the most commonly asked SQL questions in senior interviews. The difference affects storage size, performance, and whether your data can handle international characters (Unicode). Choosing the wrong type causes either wasted storage or data corruption.

Deep Explanation with Examples

char(10) — Fixed Width

Stores exactly 10 bytes always. If you store "Hi", it stores "Hi        " (padded to 10). Wastes space for shorter strings.

Use For: Codes that are always the same length — country codes ('IN', 'US'), status flags ('Y'/'N'), fixed-format IDs.

varchar(255) — Variable Width, ASCII

Stores only the actual data + 2 bytes overhead. "Hi" stores 2 bytes, "Hello World" stores 11 bytes. Efficient for variable-length English text.

Use For: English-only fields — URLs, product codes, internal identifiers, email addresses (if you're sure about language).

nvarchar(255) — Variable Width, Unicode

Like varchar but each character takes 2 bytes. Supports any language — Hindi (हिन्दी), Arabic (عربي), Chinese (中文), emoji. The 'n' stands for National (Unicode).

Use For: Any user-facing text — names, addresses, product descriptions, comments, anything a user might type in any language.

nvarchar(MAX) — Large Unicode Text

Can store up to ~2GB of text. Not stored inline in the data page — separate LOB storage. Slower for indexing.

Use For: Blog posts, article content, JSON/XML data, long descriptions. Avoid indexing nvarchar(MAX) columns.

);

A trigger is a special stored procedure that automatically fires before or after a DML event (INSERT, UPDATE, DELETE) on a table. You cannot call a trigger manually — SQL Server fires it automatically.

Magic Tables Inside a trigger, SQL Server provides two virtual tables: INSERTED (new/updated rows) and DELETED (old/deleted rows). These are your primary tool for reading what changed.

AFTER INSERT Trigger — Audit Log

-- Fires every time a row is inserted into Orders
CREATE OR ALTER TRIGGER trg_Orders_AfterInsert
ON Orders
AFTER INSERT
AS
BEGIN
    SET NOCOUNT ON;
    INSERT INTO AuditLog (TableName, Action, RecordId, ChangedAt)
    SELECT 'Orders', 'INSERT', Id, GETDATE()
    FROM INSERTED;   -- INSERTED holds the newly added row(s)
END

AFTER UPDATE Trigger — Track Price Changes

CREATE OR ALTER TRIGGER trg_Product_PriceChange
ON Product
AFTER UPDATE
AS
BEGIN
    SET NOCOUNT ON;
    -- Only fire if Price column was actually changed
    IF UPDATE(Price)
    BEGIN
        INSERT INTO PriceHistory (ProductId, OldPrice, NewPrice, ChangedAt)
        SELECT d.Id, d.Price, i.Price, GETDATE()
        FROM DELETED d                  -- old values
        INNER JOIN INSERTED i ON d.Id = i.Id   -- new values
        WHERE d.Price <> i.Price;
    END
END

BEGIN

23C SQL Performance Tuning

Performance tuning is the process of finding and fixing slow queries. The main tools are Execution Plans, indexes, and query rewriting.

Step 1 — Find the Slow Query

-- Top 10 most expensive queries by total CPU (SQL Server)
SELECT TOP 10
    qs.total_elapsed_time / qs.execution_count AS avg_elapsed_us,
    qs.execution_count,
    SUBSTRING(qt.text, (qs.statement_start_offset/2)+1, 200) AS query_text
FROM sys.dm_exec_query_stats qs
CROSS APPLY sys.dm_exec_sql_text(qs.sql_handle) qt
ORDER BY avg_elapsed_us DESC;

Step 2 — Read the Execution Plan

OperatorMeaningAction
Table ScanNo index — reads every rowAdd an index on filtered/joined column
Index ScanIndex exists but reading all rows in itAdd a more selective index or covering index
Index SeekEfficient — jumps straight to matching rowsGood — aim for this
Key LookupIndex found row but fetches extra columns from heapUse INCLUDE columns in index
Hash Join / SortMemory-heavy operationsCan indicate missing index on join column

Step 3 — Indexing Strategy

-- Covering index: OrderId is the seek key, extra columns come free
CREATE NONCLUSTERED INDEX IX_OrderItem_Covering
ON OrderItem(OrderId)
INCLUDE (ItemId, Quantity, Price);   -- eliminates Key Lookup

-- Composite index: column order matters — put equality first, range last
CREATE NONCLUSTERED INDEX IX_Orders_Customer_Date
ON Orders(CustomerId, OrderDate DESC);
-- Good for: WHERE CustomerId = 5 ORDER BY OrderDate DESC

Anti-PatternProblemFix
SELECT *Fetches unused columns, breaks covering indexSelect only needed columns
Function on column: WHERE YEAR(OrderDate) = 2024Prevents index seek — scans entire tableWHERE OrderDate >= '2024-01-01' AND OrderDate < '2025-01-01'
Implicit conversion: WHERE Id = '123' (Id is INT)Type mismatch forces scanMatch parameter type: WHERE Id = 123
Correlated subquery in SELECTRuns once per rowRewrite with JOIN or window function
NOT IN (subquery with NULLs)Returns zero rows silentlyUse NOT EXISTS instead
OR on indexed columnOften causes scanRewrite as UNION ALL

Stored Procedure — Place Order with Transaction

CREATE OR ALTER PROC PlaceOrder
    @customerId    INT,
    @purchasedItems Type_Order_Item READONLY
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRY
        IF @customerId IS NULL OR (SELECT COUNT(*) FROM @purchasedItems) < 1
            RETURN;

        DECLARE @NewOrderId INT;

        BEGIN TRANSACTION
            INSERT INTO Orders (CustomerId, OrderDate)
            VALUES (@customerId, GETDATE());

            SET @NewOrderId = SCOPE_IDENTITY();

            INSERT INTO OrderItem (OrderId, ItemId, Price, Quantity)
            SELECT @NewOrderId, ItemId, Price, Qty FROM @purchasedItems
        COMMIT TRANSACTION
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        THROW;
    END CATCH
END

HTTP Interceptor — JWT + Error Handling

// Intercept every HTTP request — add token, handle 401
@Injectable()
export class AuthInterceptor implements HttpInterceptor {
    constructor(private auth: AuthService, private router: Router) {}

    intercept(req: HttpRequest<any>, next: HttpHandler): Observable<HttpEvent<any>> {
        const token = this.auth.getToken();

        const authReq = token
            ? req.clone({ setHeaders: { Authorization: `Bearer ${token}` } })
            : req;

        return next.handle(authReq).pipe(
            catchError((err: HttpErrorResponse) => {
                if (err.status === 401) {
                    this.auth.logout();
                    this.router.navigate(['/login']);
                }
                if (err.status === 0) {
                    console.error('Network error');
                }
                return throwError(() => err);
            })
        );
    }
}

Basic Authentication

Basic Auth sends a username:password string encoded in Base64 in every HTTP request header. It is the simplest form of HTTP authentication — no tokens, no sessions. The server decodes the header and validates credentials on every single request.

Format: Authorization: Basic dXNlcjpwYXNzd29yZA==

AspectDetail
EncodingBase64 — NOT encryption. Easily decoded. Must use HTTPS.
StatelessCredentials sent on every request — no session needed
Security RiskIf HTTPS is not enforced, credentials are exposed in plaintext
Best ForInternal APIs, machine-to-machine, simple admin tools
Not ForPublic-facing apps, mobile apps, third-party access

OAuth 2.0 — What & Why

OAuth 2.0 is an authorization framework — not an authentication protocol. It allows a third-party application to access a user's resources on another service without the user sharing their password. The user grants the app a limited, scoped access token.

Real-Life Example: "Login with Google" — your app never sees the user's Google password. Google issues your app an access token with limited scope (e.g., read profile only).

OAuth 2.0 Key Roles

Resource Owner
The user — the person who owns the data and grants access.
Client
Your application that wants access (web app, mobile app).
Authorization Server
Issues tokens (Google, Azure AD, Auth0, Okta). Verifies identity.
Resource Server
The API that holds protected data. Validates the access token on each request.

33 3rd Party Integration — Stripe & Payment Gateways

Third-party payment integrations (Stripe, Razorpay, PayU, PayPal) follow a common pattern: your server never handles raw card data (PCI compliance). Instead, the payment provider's SDK tokenizes card details on the client side and your server only processes the token.

Stripe Integration Strategy — Step by Step

Step 1 — Account & Keys
Create Stripe account, get Publishable Key (frontend) and Secret Key (backend only — never expose). Store Secret Key in Azure Key Vault / env variables, never in code.
Step 2 — Frontend Tokenization
Use Stripe.js / Stripe Elements in Angular. User enters card details into Stripe's hosted iframe — you never touch raw card data. Stripe returns a PaymentMethod ID.
Step 3 — Create Payment Intent
Your .NET API calls Stripe API to create a PaymentIntent with amount + currency. Returns a client_secret to the frontend.
Step 4 — Confirm Payment
Frontend uses client_secret to confirm the payment with Stripe. Stripe handles 3DS authentication if needed.
Step 5 — Webhook Verification
Stripe sends events (payment_intent.succeeded, payment_intent.failed) to your webhook endpoint. Always verify the Stripe-Signature header to prevent spoofed events.
Step 6 — Idempotency
Use Stripe's Idempotency-Key header on all API calls to prevent duplicate charges if the request is retried due to a timeout.

35 Strategy: Designing New or Existing DB (SQL)

New Database Design Strategy

1. Domain Modeling First
List all entities from requirements. Map relationships (1:1, 1:N, M:N). Use an ERD tool (dbdiagram.io, Lucidchart) before writing SQL.
2. Normalize to 3NF
Eliminate data redundancy. Apply 1NF (atomic columns), 2NF (no partial dependency), 3NF (no transitive dependency). Denormalize only for performance with clear justification.
3. Naming Conventions
Tables: PascalCase singular (Order, not Orders). PKs: Id or OrderId. FKs: CustomerId references Customer(Id). Consistent naming = readable queries.
4. Audit Columns
Every table gets: CreatedAt DATETIME, CreatedBy INT, ModifiedAt DATETIME, ModifiedBy INT, IsDeleted BIT. Soft deletes — never hard delete production data.
5. Indexes Strategy
PK = clustered index by default. Add non-clustered on FK columns, frequently filtered columns, columns in WHERE/ORDER BY. Avoid over-indexing — each index slows INSERTs.
6. Constraints
NOT NULL where applicable. CHECK constraints for valid values (e.g. Status IN ('Active','Inactive')). UNIQUE on business keys. Foreign key constraints for referential integrity.

Existing Database — Migration & Change Strategy

Changing an existing production database requires extreme care. Downtime is often not acceptable. Always use a non-destructive, backward-compatible migration approach.

Change TypeSafe Approach
Add columnAdd as NULL first, backfill data, then add NOT NULL constraint — never add NOT NULL with no default to existing table
Rename columnAdd new column → write to both → migrate reads → drop old column (3-phase)
Change data typeTest on a copy first. For NVARCHAR length increase — safe. Decrease — risky, truncation possible.
Drop table/columnNever drop immediately. Mark as deprecated. Remove app references. Wait one release cycle. Then drop.
Large data migrationsBatch updates (10,000 rows at a time). Never UPDATE millions of rows in one transaction — lock escalation.

39 Resilience — Polly, Retry, Circuit Breaker

Polly is the standard .NET resilience library. In production, external services (payment gateways, APIs, databases) fail intermittently. Polly policies let your application recover gracefully instead of failing hard.

Core Polly Policies

Retry
Retries a failed operation N times with optional delay. Use for transient failures: network timeouts, 503 responses. Don't retry non-idempotent operations (POST creating a resource) without idempotency keys.
Circuit Breaker
After N consecutive failures, "opens" the circuit for a duration. Stops all calls to the failing service. Gives it time to recover. Prevents cascading failures. Auto-resets after the duration.
Timeout
Cancels the operation if it takes longer than X seconds. Prevents threads hanging forever on a hung downstream service.
Bulkhead Isolation
Limits concurrent calls to a service. Prevents one slow dependency from consuming all thread-pool threads and starving other services.
Fallback
Returns a default value or cached response when all retries fail. "If payment service is down, return 'Service Unavailable' instead of 500 error."
Hedge
(.NET 8 new) Sends a duplicate request if the first is too slow. First response wins. Reduces tail latency for critical paths.

            BackoffType       = DelayBackoffType.Exponential,

                { Exception: HttpRequestException } => PredicateResult.True(),

                _ => PredicateResult.False()

    });

Design: URL Shortener (Classic Interview Question)

Requirements: shorten URLs, redirect short→original, 100M URLs, 10M daily redirects, low latency.

  • Short code generation: Base62 encode a counter (a-z, A-Z, 0-9) → 7 chars = 62⁷ = 3.5 trillion unique codes
  • Storage: SQL Server — ShortCode(PK), OriginalUrl, CreatedBy, Clicks, CreatedAt
  • Redirect performance: Redis cache (shortCode → URL) with 24h TTL. Cache hit = <5ms. Cache miss = DB lookup + cache set
  • Write path: API → validate URL → generate code → save to DB → cache
  • Read path (hot): Load Balancer → API → Redis (80% hit rate) → 200ms redirect
  • Analytics: Fire-and-forget to Service Bus → background consumer increments click count

Design: Notification Service

Requirements: send email/SMS/push, 1M notifications/day, retries on failure, templated messages.

  • Producer: Any service publishes NotificationRequested event to Service Bus topic
  • Consumer: Separate Notification microservice subscribes, processes messages
  • Channel routing: Strategy pattern — EmailStrategy, SmsStrategy, PushStrategy — selected by channel type
  • Retry: Service Bus dead-letter queue after 5 failures. Alert fires for dead-letter backlog
  • Templates: Template engine (Scriban) with merge fields. Templates stored in DB, cached in Redis
  • Rate limiting: SendGrid/Twilio rate limits handled by Polly + token bucket algorithm

CAP Theorem — Must Know

PropertyMeaningExample
ConsistencyAll nodes return the same data at the same timeSQL Server with synchronous replication
AvailabilityEvery request gets a (possibly stale) responseDNS — always responds, may be cached
Partition ToleranceSystem works even if network splits nodesRequired for any distributed system

In a distributed system, partition tolerance is mandatory (networks fail). So the real choice is: CP (consistency + partition tolerance — sacrifice availability) or AP (availability + partition tolerance — sacrifice consistency).

Real Example: Bank transfers need CP — you can never show wrong balance. Social media likes are fine as AP — a slight lag before all users see the updated count is acceptable.

43 Scalability Strategies

Horizontal Scaling Checklist

For your .NET API to scale horizontally (multiple instances behind a load balancer), it must be stateless. Any state must live outside the process.

Session State → Redis
HTTP sessions must go in Redis distributed cache, not in-memory. Otherwise user A's session is on Server 1 and Server 2 doesn't know about it.
File Uploads → Blob Storage
Never save files to local disk. Use Azure Blob Storage / S3. All instances access the same storage account.
Background Jobs → External Queue
Use Azure Service Bus or Hangfire with SQL/Redis backend. In-memory queues are lost when an instance restarts.
SignalR → Azure SignalR Service
WebSocket connections are sticky to one server. Use Azure SignalR Service to backplane — messages route across all instances.
Distributed Locks → Redis
Use Redlock (Redis distributed lock) when you need exclusive access across instances (e.g., run a cron job only once across 5 API instances).
Config → Azure App Config
Environment-specific config centralized. Feature flags change without redeploy. All instances pick up changes in real time.

// Azure Service Bus — Consumer (Email notification service)
public class EmailNotificationConsumer : BackgroundService
{
    protected override async Task ExecuteAsync(CancellationToken ct)
    {
        var processor = _serviceBusClient.CreateProcessor("order-events", "email-subscription");
        processor.ProcessMessageAsync  += HandleMessageAsync;
        processor.ProcessErrorAsync    += HandleErrorAsync;
        await processor.StartProcessingAsync(ct);
        await Task.Delay(Timeout.Infinite, ct);
    }

    private async Task HandleMessageAsync(ProcessMessageEventArgs args)
    {
        var orderEvent = JsonSerializer.Deserialize<OrderPlacedEvent>(args.Message.Body);
        await _emailService.SendOrderConfirmationAsync(orderEvent);
        await args.CompleteMessageAsync(args.Message);  // acknowledge — remove from queue
    }

    private Task HandleErrorAsync(ProcessErrorEventArgs args)
    {
        _logger.LogError(args.Exception, "Service Bus error: {Source}", args.ErrorSource);
        return Task.CompletedTask;  // message auto-retried by Service Bus
    }
}

Key Message Queue Concepts

Dead Letter Queue (DLQ)
Messages that fail after max retries go to the DLQ. Monitor DLQ in production — it tells you what's breaking silently. Alert when DLQ count > 0.
Idempotent Consumer
Service Bus may deliver a message more than once (at-least-once delivery). Always check if you've already processed the message ID before processing.
Message TTL
Set message expiry (Time To Live). A payment retry message older than 24 hours is useless — expired messages go to DLQ automatically.
Competing Consumers
Multiple instances of a consumer service all listen to the same queue. Service Bus delivers each message to only ONE consumer — auto load balancing.

Essential RxJS Operators

// switchMap — cancel previous, use latest (search autocomplete)
this.searchControl.valueChanges.pipe(
  debounceTime(300),           // wait 300ms after typing stops
  distinctUntilChanged(),      // ignore same value repeated
  switchMap(query =>           // cancel previous HTTP call, start new one

      catchError(() => of([]))   // return empty array on error

from(files).pipe(
  mergeMap(file => this.uploadService.upload(file))

// concatMap — run sequentially (order matters)
from(orderIds).pipe(
  concatMap(id => this.orderService.process(id))  // one at a time, in order

// combineLatest — combine latest from multiple streams
combineLatest([this.user$, this.permissions$]).pipe(
  map(([user, perms]) => ({ ...user, canEdit: perms.includes('edit') }))

// takeUntil — auto-unsubscribe on component destroy (prevents memory leaks)
private destroy$ = new Subject<void>();

29 Common Interview Traps

⚠ Trap 1 Scoped inside Singleton — captive dependency causes stale data or runtime errors. Use IServiceScopeFactory.
⚠ Trap 2 IEnumerable for DB filtering — loads entire table into memory, then filters in C#. Use IQueryable.
⚠ Trap 3 Missing await — method returns a Task that nobody awaits; exception swallowed silently.
⚠ Trap 4 async void — exceptions are unobservable and crash the process. Always use async Task.
⚠ Trap 5 SELECT * — fetches unnecessary columns, breaks if schema changes. Always select specific columns.
⚠ Trap 6 No CancellationToken — long operations run even after client disconnects. Always accept and pass CancellationToken.
⚠ Trap 7 Lazy loading in APIs — causes N+1 queries silently, only noticed under load. Use eager loading with Include().
